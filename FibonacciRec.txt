program fibonacciRec;
vars
int i, input, output;

function int fib(int n)
vars 
int uno, dos;
{
    if (n <= 1) then {
        return(n);
    }
    else{
        uno = fib(n - 1);
        dos = fib(n - 2);
        return(uno + dos);
    }
}

main(){
    read(input);
    for i = 0 to input do {
        output = fib(input);
        write("pos ", i ," is ", output);
    }
}
