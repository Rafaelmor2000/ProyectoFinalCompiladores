program area;
vars
int ancho, largo, otro;
int alto[5];
int i, algo;

main()
{
    ancho = 5;
    read(otro, largo);
    largo = 1 * 2 * 3;
    for i = 0 to 5 do {
        alto[i] = i;
        write(alto[i]);
    }

    write("el area del rectangulo es: ", ancho / largo > 2 * (ancho + largo) & true);
}
